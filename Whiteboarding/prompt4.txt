Question #4: Checking for uniquenessness
Write an algorithm that determines whether all the elements in a string are uniqueness. You may not convert the string into an array or use array methods to solve this problem. The algorithm should return a boolean.

Example
Input: "hello"
Output: false

Input: "copyright"
Output: true


      function uniqueness(string) {
        let notUnique = "";     //empty string

        for(let i = 0; i < string.length; i++) {  //loop to check if character is in string
          if(notUnique.includes(string[i]) === false) { // if charcter is in string = false
            notUnique += string[i];                     // push to notUnique 
          }
        }

        if (notUnique === string) {    // compare input string to notUnique
          return true;                 // same then return true
        } else {                        
          return false;                // else return false
        }
      }
